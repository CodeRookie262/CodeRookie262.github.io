    # Vue双向数据绑定原理及实现

    ​ 说到Vue,很多人应该就会想到数据双向绑定吧.

    ​ 但是,Vue 的数据双向绑定的原理就是利用 Object.defineProperty来属性的,监听data对象的每个属性,一旦数据改变就会驱动视图层的改变;

    ​ 那么又是怎么利用 Object.defineProperty来监听和实现数据驱动视图呢?以下是一个简单的小案例

    HTML Code

    <p id="con"></p>
<input type="text" id="ipt" />
    1
    2

    JavaScript Code

    var ipt = document.getElementById("ipt"),
    con = document.getElementById("con");
//data 对象创建
var data = {
    value: ""
},
middle = null;
ipt.oninput = function (){
    data.value = this.value;
}
Object.defineProperty(data,"value",{
    //获取 data.value 时触发
    get: function(){
        return middle;
    },
    //设置 data.value 时触发
    set: function(newValue){
        middle = newValue;
        con.innerHTML = middle;
        return middle;
    }
})

    # 初始化函数的封装 init

    Wz_Vue.prototype.init = function (option){
    this.$data = option.data; //用户传入 data 对象
    this.$el = option.el; //实例化所绑定的标签节点
    this.$methods = option.methods; //用户传入的事件对象
    this.mv_bind = {}; //这个是专门接收用户的绑定属性或事件及其节点信息的
    //对 data 的属性进行双向绑定监听
    this.monitor(this.$data)
    //视图更新
	this.complie(this.$el)
}

    # 数据双向绑定处理函数封装 monitor

    Wz_Vue.prototype.monitor = function (data){
    for(var key in data){
        var target = data[key];
        //判断 target 是否依然是一个对象,是的话则递归处理
        if(target instanceof Object){
            this.monitor(target);
        }
        // 开启 model 与 view 的映射设置,作用就是当 model 的数据改变,就会驱动 view 的改变 ====> 数据驱动
        this.mv_bind[key] = {
            set: []
        }
        var bind = this.mv_bind[key];
        //对 data 的属性进行双向监听
        Object.defineProperty(data,key,{
            enumerable: true,
            get: function(){
                return target;
            },
            set: function(newValue){
                if(target != newValue){
                    target = newVaule;
                    //更新视图 
                    bind.set.forEach(function(item){
                            item.updata()
                        });
                }
                return target;
            }
        })
    }
}

    # 节点遍历及其重新渲染 complie

    complie(root){//root 为 myVue的 $el 绑定的最外层节点 
	let nodeList = root.children;
	// 遍历每个节点
	[...nodeList].forEach(item => {
		var node = item;
		// 事件库监听绑定
		EventLibrary
			.forEach(eventName => {
				let name = `wz-${eventName}`;
				if(node.hasAttribute(name)){
					//截取绑定函数名称
					var fnName = node.getAttribute(name);
					// 函数存在性及其类型判断
					if(fnName){
						if(this.$methods[fnName] && typeof this.$methods[fnName] == "function"){
							node["on" + eventName] = this.$methods[fnName].bind(this.$data);
					}else if(this.$methods[fnName]){
						EventWarn(node,eventName,`${fnName}不是一个函数,请在methods对象修改该属性`)
					}else{
						EventWarn(node,eventName,`${fnName}函数未注册,请在methods对象注册该事件`)
						}
					}
								
				}
		})
		//wz-model  表单动态改变数据 &视图
		let model = node.hasAttribute("wz-model");
		if(model && /input|textarea/gi.test(node.tagName)){
		let attrValue = node.getAttribute("wz-model");
		// console.log(attrValue);
		this.mv_bind[attrValue].set.push(new Watcher(node,this,attrValue,"value"))
		//输入事件触发视图更新
			node.addEventListener("input",() => {
				this.$data[attrValue] = node["value"];
				})
			}
			if(node.getAttribute("wz-bind")){
				let attrValue = node.getAttribute("wz-bind");
				// console.log(node,attrValue)
				this.mv_bind[attrValue].set.push(new Watcher(node,this,attrValue,"innerHTML"))
				}
				//如果该节点不是底层元素则递归遍历内部节点 
				node.children.length && this.complie(item)
			})
		}
	}

    # 观察者 Watcher

    function Watcher(ele,vm,vmProp,attrProp){
    Object.assign(this,{ele,vm,vmProp,attrProp})
    // 绑定 DOM 属性
    this.updata()
}
Watcher.prototype.updata = function(){
    let {ele,vm,vmProp,attrProp} = this;
    ele[attrProp] = vm.$data[vmProp];
}
